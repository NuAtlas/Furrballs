21FurrBall* FurrBall::CreateBall(std::string DBpath, size_t PageSize, size_t numPages, bool overwrite) noexcept
22{
23 rocksdb::Options options;
24 rocksdb::DB* db;
25 options.compression = rocksdb::kLZ4Compression;
26 //fb.options.OptimizeForPointLookup();
27 options.create_if_missing = true;
28 rocksdb::Status status =
29 rocksdb::DB::Open(options, DBpath, &db);
30 if (!status.ok()) {
31 //ERR
32 return nullptr;
33 }
34 //Setup Cache.
35 size_t availMem = MemoryManager::GetAvailableMemory();
36 if (availMem < PageSize * numPages) {
37 //We don't have enough memory to allocate all the pages.
38 while ((--numPages) || availMem < PageSize * numPages);
39 if (numPages <= 0) {
40 Logger::getInstance().error("Not enough memory");
41 return nullptr;
42 }
43 }
44 ARCPolicy<size_t, void*> Cache(numPages);
45 //Allocate Slab.
46 char* slab = static_cast<char*>(malloc(PageSize * numPages));
47 if (!slab) {
48 Logger::getInstance().warning("Could not allocated memory slab.");
49 //Maybe attempt to allocate fragmented slab.
50 //for now return nullptr
51 return nullptr;
52 }
53 size_t PagePointer = 0;
54 for (int i = 0; i < numPages; i++) {
55 Cache.add(PagePointer, slab + PagePointer);
56 }
57 FurrBall* fb = new FurrBall();
58 fb->DataMembers->db = db;
59 return fb;
60}
