56 static bool ProtectMemory(void* buffer, const size_t& size) {
57 std::lock_guard<std::mutex> lock(ProtectMutex);
58#ifdef _WIN32
59 DWORD oldProtect;
60 if (!VirtualProtect(buffer, size, PAGE_READWRITE, &oldProtect)) {
61 Logger::getInstance().error("Failed to set memory protection on Windows");
62 return false;
63 }
64#else
65 if (mprotect(buffer, size, PROT_READ | PROT_WRITE) != 0) {
66 Logger::getInstance().error("Failed to set memory protection on Linux");
67 return false;
68 }
69#endif
70 return true;
71 }

100 /**
101 * @brief Attempts to allocate increasingly larger blocks of memory until it fails, then returns the size of the largest successful allocation.
102 */
103 static size_t GetLargestContiguousBlock() {
104 size_t step = 1024 * 1024; // Start with 1MB
105 size_t maxBlockSize = 0;
106 void* buffer = nullptr;
107
108 while (true) {
109 buffer = AllocateMemory(maxBlockSize + step);
110 if (buffer) {
111 FreeMemory(buffer);
112 maxBlockSize += step;
113 }
114 else {
115 break;
116 }
117 }
118
119 return maxBlockSize;
120 }

124 * @brief Implements the ARC eviction policy
125 * TODO: Implement Adaptive Memory Pooling (AMP)

127 * @see S3FIFOPolicy

131 template<class Key, class Value>
132 class ARCPolicy {
133 public:
134 typedef void(*EvictionCallback)(Key&);
135 private:
136 std::list<Key> t1; // Recently added
137 std::list<Key> t2; // Recently used
138 std::list<Key> b1; // Ghost entries for t1
139 std::list<Key> b2; // Ghost entries for t2
140 std::unordered_map<Key, Value> map; // Key to value mapping
141 std::list<Key> Window; // t1 + t2 (the cache)
142 std::list<Key> GhostEntries; // b1 + b2 Ghost entries
143 size_t capacity;
144 size_t p; // Target size for t1
145 EvictionCallback evictionCallback = []() {};//NO-OP by default.
146
147 void replace(const Key& key) {
148 if (!t1.empty() && (t1.size() > p || (std::find(b2.begin(), b2.end(), key) != b2.end() && t1.size() == p))) {
149 // Move from t1 to b1
150 auto old = t1.back();
151 t1.pop_back();
152 b1.push_front(old);
153 map.erase(old);
154 }
155 else {
156 // Move from t2 to b2
157 auto old = t2.back();
158 t2.pop_back();
159 b2.push_front(old);
160 map.erase(old);
161 }
162 }
163
164 void evict() {
165 evictionCallback();
166 if (t1.size() + b1.size() >= capacity) {
167 if (t1.size() < capacity) {
168 evictionCallback(b1.back())
169 b1.pop_back();
170 }
171 else {
172 evictionCallback(t1.back())
173 t1.pop_back();
174 }
175 }
176 if (t1.size() + t2.size() + b1.size() + b2.size() >= 2 * capacity) {
177 if (t2.size() + b2.size() > capacity) {
178 evictionCallback(b2.back())
179 b2.pop_back();
180 }
181 else {
182 evictionCallback(t2.back())
183 t2.pop_back();
184 }
185 }
186 }
187
188 public:
189 /**
190 * @brief Creates a cache following ARC policy.
191 * @param cap Capacity of the cache.
192 */
193 ARCPolicy(size_t cap) : capacity(cap), p(1) {}
194
195 void setEvictionCallback(EvictionCallback cb) {
196 evictionCallback = cb;
197 };
198 /**
199 * @return true if the key exists.
200 */
201 bool contains(const Key& key) const {
202 return map.find(key) != map.end();
203 }
204 /**
205 * @brief Promotes a Key.
206 */
207 void touch(const Key& key) {
208 if (std::find(t1.begin(), t1.end(), key) != t1.end()) {
209 t1.remove(key);
210 t2.push_front(key);
211 }
212 else if (std::find(t2.begin(), t2.end(), key) != t2.end()) {
213 t2.splice(t2.begin(), t2, std::find(t2.begin(), t2.end(), key));
214 }
215 else if (std::find(b1.begin(), b1.end(), key) != b1.end()) {
216 // Case when the key is in b1
217 p = std::min(capacity, p + std::max(b2.size() / b1.size(), 1UL));
218 replace(key);
219 b1.remove(key);
220 t2.push_front(key);
221 map[key] = Value(); // Assuming default constructor exists
222 }
223 else if (std::find(b2.begin(), b2.end(), key) != b2.end()) {
224 // Case when the key is in b2
225 p = std::max(0, static_cast<int>(p) - std::max(b1.size() / b2.size(), 1UL));
226 replace(key);
227 b2.remove(key);
228 t2.push_front(key);
229 map[key] = Value(); // Assuming default constructor exists
230 }
231 }
232 /**
233 * @brief Adds a Key-Value Pair the the cache.
234 */
235 void add(const Key& key, const Value& value) {
236 if (map.size() >= capacity) {
237 evict();
238 }
239 t1.push_front(key);
240 map[key] = value;
241 }
242 /**
243 * @brief Gets a value from the cache.
244 */
245 Value get(const Key& key) {
246 touch(key);
247 return map[key];
248 }
249 /**
250 * @brief Changes a value if it exsits or adds it.
251 */
252 void set(const Key& key, const Value& value) {
253 if (contains(key)) {
254 map[key] = value;
255 touch(key);
256 }
257 else {
258 add(key, value);
259 }
260 }
261 };
