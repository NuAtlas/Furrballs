103 /**
104 * @brief Implements the ARC eviction policy
105 *
106 * You can create and manage your own cache separately by instantiating a Policy object and using it.
107 * @see S3FIFOPolicy
108 * @see LRUPolicy
109 * @see LFUPolicy
110 */
111 template<class Key, class Value>
112 class ARCPolicy {
113 private:
114 std::list<Key> t1; // Recently added
115 std::list<Key> t2; // Recently used
116 std::list<Key> b1; // Ghost entries for t1
117 std::list<Key> b2; // Ghost entries for t2
118 std::unordered_map<Key, Value> map; // Key to value mapping
119 size_t capacity;
120 size_t p; // Target size for t1
121
122 void replace(const Key& key) {
123 if (!t1.empty() && (t1.size() > p || (std::find(b2.begin(), b2.end(), key) != b2.end() && t1.size() == p))) {
124 // Move from t1 to b1
125 auto old = t1.back();
126 t1.pop_back();
127 b1.push_front(old);
128 map.erase(old);
129 }
130 else {
131 // Move from t2 to b2
132 auto old = t2.back();
133 t2.pop_back();
134 b2.push_front(old);
135 map.erase(old);
136 }
137 }
138
139 void evict() {
140 if (t1.size() + b1.size() >= capacity) {
141 if (t1.size() < capacity) {
142 b1.pop_back();
143 }
144 else {
145 t1.pop_back();
146 }
147 }
148 if (t1.size() + t2.size() + b1.size() + b2.size() >= 2 * capacity) {
149 if (t2.size() + b2.size() > capacity) {
150 b2.pop_back();
151 }
152 else {
153 t2.pop_back();
154 }
155 }
156 }
157
158 public:
159 /**
160 * @brief Creates a cache following ARC policy.
161 * @param cap Capacity of the cache.
162 */
163 ARCPolicy(size_t cap) : capacity(cap), p(0) {}
164 /**
165 * @return true if the key exists.
166 */
167 bool contains(const Key& key) const {
168 return map.find(key) != map.end();
169 }
170 /**
171 * @brief Promotes a Key.
172 */
173 void touch(const Key& key) {
174 if (std::find(t1.begin(), t1.end(), key) != t1.end()) {
175 t1.remove(key);
176 t2.push_front(key);
177 }
178 else if (std::find(t2.begin(), t2.end(), key) != t2.end()) {
179 t2.splice(t2.begin(), t2, std::find(t2.begin(), t2.end(), key));
180 }
181 else if (std::find(b1.begin(), b1.end(), key) != b1.end()) {
182 // Case when the key is in b1
183 p = std::min(capacity, p + std::max(b2.size() / b1.size(), 1UL));
184 replace(key);
185 b1.remove(key);
186 t2.push_front(key);
187 map[key] = Value(); // Assuming default constructor exists
188 }
189 else if (std::find(b2.begin(), b2.end(), key) != b2.end()) {
190 // Case when the key is in b2
191 p = std::max(0, static_cast<int>(p) - std::max(b1.size() / b2.size(), 1UL));
192 replace(key);
193 b2.remove(key);
194 t2.push_front(key);
195 map[key] = Value(); // Assuming default constructor exists
196 }
197 }
198 /**
199 * @brief Adds a Key-Value Pair the the cache.
200 */
201 void add(const Key& key, const Value& value) {
202 if (map.size() >= capacity) {
203 evict();
204 }
205 t1.push_front(key);
206 map[key] = value;
207 }
208 /**
209 * @brief Gets a value from the cache.
210 */
211 Value get(const Key& key) {
212 touch(key);
213 return map[key];
214 }
215 /**
216 * @brief Changes a value if it exsits or adds it.
217 */
218 void set(const Key& key, const Value& value) {
219 if (contains(key)) {
220 map[key] = value;
221 touch(key);
222 }
223 else {
224 add(key, value);
225 }
226 }
227 };
